function finalizar() { const obs = document.getElementById('obs').value.trim(); if (!obs && materialesSeleccionados.length === 0) { alert('Agregá una observación o seleccioná tareas antes de finalizar.'); return; } const obsFinal = obs || '(sin observaciones adicionales)'; const materialFinal = materialesSeleccionados.join(', '); clearInterval(timerIv); localStorage.removeItem('isusa_trabajo'); const fin = new Date(); const min = Math.round(timerSec / 60); const idx = DENUNCIAS.findIndex(x => String(x.id) === String(trabajo.id)); if (idx >= 0) { const key = String(trabajo.id); operariosActivos[key] = (operariosActivos[key] || []).filter(o => o !== sesion.operario); } const payload = { accion: 'finalizar_operario', num_ot: trabajo.num_ot, operario: sesion.operario, hora_fin: fmtH(fin), duracion_min: min, observaciones: obsFinal, material: materialFinal }; const imgFin = new Image(); imgFin.src = SCRIPT_DENUNCIAS + '?payload=' + encodeURIComponent(JSON.stringify(payload)); const parte = { accion: 'parte', num_ot: trabajo.num_ot, fecha: trabajo.fecha, planta: trabajo.planta, equipo: trabajo.equipo, tipo_trabajo: trabajo.tipo, prioridad: trabajo.prioridad, operario: sesion.operario, taller: sesion.taller, hora_inicio: fmtH(trabajo.horaInicio), hora_fin: fmtH(fin), duracion_min: min, duracion_hs: (timerSec/3600).toFixed(2), observaciones: obsFinal, material: materialFinal }; const imgParte = new Image(); imgParte.src = SCRIPT_PARTES + '?payload=' + encodeURIComponent(JSON.stringify(parte)); document.getElementById('ok-txt').innerHTML = `${trabajo.equipo} — ${trabajo.planta}
